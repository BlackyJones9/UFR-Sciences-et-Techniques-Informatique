Algorithme HDS en Langage Algorithmique : 

Structure Point
	Tableau[1,2] de réels compris entre 0 et 1

Fonction demiSomme(D : Tableau[1..n] de Tableau[1..n] de réels, i : Entier)
    var min1er, min2nd, : Réels, j : Entier
    Début
    min1er <- inf
    min2nd <- inf
    Pour j allant de 0 à n  #On cherche les deux arêtes de poids minimal reliés au sommet i
        Début
        Si non(i == j)
            Début
            Si D[i][j] <= min1er
                Début
                min2nd <- min1er
                min1er <- D[i][j]
                FinSi
            Sinon Si (D[i][j] != min1er) et (D[i][j] <= min2nd)
                Début
                min2nd <- D[i][j]
                FinSinon
            FinSi
        FinPour
    renvoyer (min1er, min2nd)
    Fin


La procédure MettreAFaux met à Faux tous les éléments de la liste données en argument
La procédure MettreAMoinsUn met à -1 tous les éléments de la liste données en argument
La fonction Arrondi renvoie un arrondi au supérieur entier du réel donné en argument

Fonction HDSCalcul (D : Tableau[1..n] de Tableau[1..n] de réels, branche, poids : Réel, chemin : Tableau[1..n+1] d'Entier, visited : Tableau[1..n] de Booléens, longueurCircuit : réel, circuit : Tableau[1..n+1] d'Entiers)
    var res, branche_temp, curr_min1er, curr_min2nd, min1er : Réel, i, j : Entier
    Début
    Si hauteur == n  #Cas d'arrêt de la récursivité
        Début
        Si D[chemin[hauteur]][chemin[0]] != 0  #On vérifie que le chemin n'est pas une boucle
            Début
            res <- poids + D[chemin[hauteur]][chemin[0]]  #On met à jour le poids afin d'ajouter la dernière arête 
            Si res < longueurCircuit  #Si l'ajout de la dernière arête est avantageux alors on contruit le circuit et on le renvoie
                Début
                circuit[:n + 1] <- chemin[:]  #On remplace les n premiers éléments par chemin
                circuit[n] <- chemin[0]
                longueurCircuit <- res
                FinSi
            FinSi
            renvoyer longueurCircuit, circuit  #On renvoie la longueur du circuit et le circuit en lui-même
        FinSi
    Pour i allant de 1 à n  #On boucle sur tous les sommets
        Début
        Si D[chemin[hauteur]][chemin[0]] != 0 et non(visited[i])  #On vérifie si le sommet i est relié au dernier sommet du chemin et si il n'a pas déjà été visité
            Début
            branche_temp <- branche  #On sauvegarde la branche pour pouvoir la modifier
            poids <- poids + D[chemin[hauteur]][i]  #On suppose que i est le prochain sommet du circuit et donc on ajoute le poids de l'arête

            curr_min1er, curr_min2nd <- demiSomme(D, chemin[hauteur])  #On cherche le 1er et le 2eme poids minimum du dernier sommet du chemin
            min1er, _ <- demiSomme(D, i)  #On cherche le 1er et le 2eme poids minimum du sommet i
            Si hauteur == 1  #Si la hauteur de la branche vaut 1 alors on soustrait la moyenne des premiers minimums
                Début
                branche <- branche - ((curr_min1er + min1er) / 2)
                FinSi
            Sinon  #Sinon il y a au moins 2 points donc on soustrait la moyenne du 1er minimum du sommet du chemin et le 2ème minimum de i
                Début
                branche <- branche - ((curr_min2nd + min1er) / 2)
                FinSinon

            Si branche + poids < longueurCircuit  #Appel récursif si ajouter le sommet i au chemin est avantageux par rapport au circuit courant
                Début
                chemin[hauteur] <- i 
                visited[i] <- Vrai  #Mise à jour du chemin et du tableau visited
                longueurCircuit, circuit <- HDSCalcul(D, branche, poids, hauteur + 1, chemin, visited, longueurCircuit, circuit)  #Appel récursif = on avance dans la recherche d'une solution
                FinSi
            
            poids <- poids - D[chemin[hauteur]][i]  #On prépare la boucle suivante en annulant les mises à jours ayant eu lieu durant la boucle
            branche <- branche_temp
            MettreAFaux(visited)  
            Pour j allant de 1 à hauteur  #On redéfinit le tableau visited afin d'assurer de ne pas passer des circuit possibles
                Début
                visited[chemin[j]] <- Vrai
                FinPour
            FinSi
        FinPour  
    renvoyer (longueurCircuit, circuit)  #On renvoie la longueur du circuit et le circuit en lui-même  
    Fin

La fonction sommetsEnPoints renvoie une liste de points selon une liste de sommets et une liste de points de référence
La fonction calculateDistance calcule la distance totale du circuit selon une matrice de distance

Fonction HDS (D : Tableau[1..n] de Tableau[1..n] de réels, pointList : Tableau[1..n] de Points)
    var chemin : Tableau[1..n+1] d'Entier, visited : Tableau[1..n] de Booléens, i, racine : Entier, dist, min1er, min2nd : Réel
    Début
    MettreAMoinsUn(chemin)
    chemin[0] <- 0  #On initialise les données utilisées par HDSCalcul
    MettreAFaux(visited)
    visited[0] <- Vrai

    dist <- 0
    Pour i allant de 1 à n
        Début
        min1er, min2nd <- demiSomme(D, i)
        dist <- dist + min1er + min2nd
        FinPour
    racine <- Arrondi(dist / 2)  #On calcul la valeur de la racine
    longueurCircuit <- inf
    MettreAMoinsUn(circuit)  #On initialise les résultats par défaut
    longueurCircuit, circuit <- HDSCalcul(D, racine, 0, 1, chemin, visited, longueurCircuit, circuit)  #On renvoie le résultat de l'appel à HDSCalcul
    circuit <- sommetsEnPoints(circuit, pointList)
    longueurCircuit <- calculateDistance(D, circuit, pointList)
    décroiser(D, longueurCircuit, circuit, pointList)
    Fin
